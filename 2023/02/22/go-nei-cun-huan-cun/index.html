Go内存缓存

1.背景

基本上每个项目都有加快服务响应或复杂计算的需求。简单快速的解决方案是使用缓存。通常，有Redis或Memcached，但我们在单实例微服务中不需要使用它们。有时候在你的Go应用程序中使用一个简单的内存缓存会更好，今天我想介绍实现内存缓存的方法。

2.内存存储方式

通用结构体定义

package cacheModel

import "errors"

type User struct {
    Id    int64  `json:"id"`
    Email string `json:"email"`
}

var (
    ErrUserNotInCache = errors.New("the user is not in cache")
)

2.1 Map方式

通常，使用map存储结构体。此外，还需要监控元素的过期时间和缓存大小。

package main

import (
  "awesomeProject/cacheModel"
  "fmt"
  "sync"
  "time"
)

// 实际存到map中的结构体内容
type cachedUser struct {
  cacheModel.User
  expireAtTimestamp int64
}

// 创建缓存结构体，包含对象的增加、删除等方法
type localCache struct {
  stop  chan struct{}
  wg    sync.WaitGroup
  mu    sync.RWMutex         // 对写锁
  users map[int64]cachedUser // map存储用户信息
}

// 创建缓存对象，cleanupInterval设置定期清除缓存过期元素
func newLocalCache(cleanupInterval time.Duration) *localCache {
  lc := &localCache{
    users: make(map[int64]cachedUser),
    stop:  make(chan struct{}),
  }
  // 启动携程后定期清除过期元素
  lc.wg.Add(1)
  go func(cleanupInterval time.Duration) {
    defer lc.wg.Done()
    lc.cleanupLoop(cleanupInterval)
  }(cleanupInterval)
  return lc
}

func (lc *localCache) cleanupLoop(interval time.Duration) {
  t := time.NewTicker(interval)
  defer t.Stop()
  for {
    select {
    case <-lc.stop: // 主动退出
      return
    case <-t.C:
      lc.mu.Lock()
      for uid, cu := range lc.users {
        if cu.expireAtTimestamp <= time.Now().Unix() {
          delete(lc.users, uid)
        }
      }
      lc.mu.Unlock()
    }
  }
}

func (lc *localCache) stopCleanup() {
  close(lc.stop)
  lc.wg.Wait()
}

// 更新元素过期时间
func (lc *localCache) update(u cacheModel.User, expireAtTimestamp int64) {
  lc.mu.Lock()
  defer lc.mu.Unlock()
  lc.users[u.Id] = cachedUser{
    User:              u,
    expireAtTimestamp: expireAtTimestamp,
  }
}

// 读缓存
func (lc *localCache) read(id int64) (cacheModel.User, error) {
  lc.mu.RLock()
  defer lc.mu.RUnlock()
  cu, ok := lc.users[id]
  if !ok {
    return cacheModel.User{}, cacheModel.ErrUserNotInCache
  }
  return cu.User, nil
}

// 删除缓存元素
func (lc *localCache) delete(id int64) {
  lc.mu.Lock()
  defer lc.mu.Unlock()
  delete(lc.users, id)
}

func main() {
  lc := newLocalCache(1 * time.Second)
  user1 := cacheModel.User{
    Id:    1,
    Email: "12345@163.com",
  }
  user2 := cacheModel.User{
    Id:    2,
    Email: "qweret@163.com",
  }
  lc.users = map[int64]cachedUser{
    1: {User: user1, expireAtTimestamp: 1676361777},
    2: {User: user2, expireAtTimestamp: 1677035802},
  }
  for {
    u1, err := lc.read(1)
    if err != nil {
      fmt.Printf("Error reading:%v", err)
      return
    }
    fmt.Println(u1.Email)
    time.Sleep(1 * time.Second)
  }
}

上面的例子我们使用用户ID作为缓存元素的Key。使用map，所有update/read/delete操作时间复杂度都是O(1)

  • 优点
    • 实现简单
    • 性能高
  • 缺点
    • 存储每一类结构体都需要实现缓存
    • 需要单独测试缓存
    • 单独的bug修复

2.2 gCache库

gCache库对缓存实现进行抽象，包含各种配置。例如，可以很简单地设置缓存淘汰规则，缓存元素最大长度，过期时间TTL等

package main

import (
  "awesomeProject/cacheModel"
  "errors"
  "fmt"
  "github.com/bluele/gcache"
  "time"
)

type gCache struct {
  Users gcache.Cache // 该对象可以缓存任何类型数据
}

const (
  cacheSize = 1_000_000
  cacheTTL  = 1 * time.Hour
)

// 创建缓存对象，使用ARC算法淘汰缓存元素
func newGCache() *gCache {
  return &gCache{
    Users: gcache.New(cacheSize).Expiration(cacheTTL).ARC().Build(),
  }
}

// 更新缓存元素过期时间
func (gc *gCache) update(u cacheModel.User, expireIn time.Duration) error {
  return gc.Users.SetWithExpire(u.Id, u, expireIn)
}

// 读取缓存
func (gc *gCache) read(id int64) (cacheModel.User, error) {
  val, err := gc.Users.Get(id)
  if err != nil {
    if errors.Is(err, gcache.KeyNotFoundError) {
      return cacheModel.User{}, cacheModel.ErrUserNotInCache
    }

    return cacheModel.User{}, fmt.Errorf("get: %w", err)
  }

  return val.(cacheModel.User), nil
}

// 删除缓存元素
func (gc *gCache) delete(id int64) {
  gc.Users.Remove(id)
}

func main() {
  g := newGCache()
  u1 := cacheModel.User{
    Id:    1,
    Email: "1234@qq.com",
  }
  if err := g.update(u1, 2*time.Second); err != nil {
    fmt.Println("update error:", err)
  }
  for {
    v, err := g.read(1)
    if err != nil {
      fmt.Println(err)
      return
    }
    fmt.Println(v)
    time.Sleep(1 * time.Second)
  }
}

结果：两秒后查询缓存，发现缓存对象已经被清理了

{1 1234@qq.com}
{1 1234@qq.com}
the user is not in cache
  • 优点
    • 可直接投入生产环境中使用
    • 接口适用任意类型
    • 不同的缓存淘汰算法：LRU，LFU，ARC
  • 缺点
    • 去缓存都需要做类型转换性能差
    • 这个库有一点时间没有维护

2.3 BigCache库

BigCache库高性能、支持并发、缓存淘汰，可存储大量元素而不影响性能。BigCache将元素放在堆中忽略GC
eg:

package main

import (
    "awesomeProject/cacheModel"
    "context"
    "encoding/json"
    "errors"
    "fmt"
    "github.com/allegro/bigcache"
    "github.com/spf13/cast"
    "time"
)

type bigCache struct {
    users *bigcache.BigCache
}

func newBigCache() (*bigCache, error) {
    ctx := context.Background()
    bCache, err := bigcache.New(ctx, bigcache.Config{
        // 分片数量（必须是2的幂次方）
        Shards: 1024,

        // 存活时间，过了该时间才会删除元素
        LifeWindow: 5 * time.Second,

        // 删除过期元素的时间间隔（清理缓存）
        // 如果设置为<=0,则不执行任何操作
        // 设置为<1秒会适得其反 bigCache只能精确到1秒
        CleanWindow: 2 * time.Second,

        // reps * lifeWindow, 仅用于初始内存分配
        MaxEntriesInWindow: 1000 * 10 * 60,

        // 以字节为单位的元素大小最大值，仅在舒适内存分配时使用
        MaxEntrySize: 500,

        // 打印内存分配信息
        Verbose: false,

        // 缓存分配的内存不会超过这个限制，MB单位
        // 如果达到值，则可以为新条目覆盖旧的元素
        // 0值表示没有限制
        HardMaxCacheSize: 256,

        // 当最旧的元素由于过期时间或没有剩余空间而被删除时，触发回调
        // 对于新元素，或者因为调用了delete。将返回一个表示原因的位掩码.
        // 默认值为nil，这意味着没有回调.
        OnRemove: nil,

        // OnRemoveWithReason当因为过期时间或没有空间时，最老一条元素被删除会触发该回调。会返回删除原因。
        // 默认值为nil。
        OnRemoveWithReason: nil,
    })
    if err != nil {
        return nil, fmt.Errorf("new big cache : %v", err)
    }
    return &bigCache{
        users: bCache,
    }, nil
}

// 更新缓存中的元素
func (b *bigCache) update(u cacheModel.User) error {
    bs, err := json.Marshal(&u)
    if err != nil {
        return fmt.Errorf("marshal:%v", err)
    }
    return b.users.Set(cast.ToString(u.Id), bs)
}

// 删除缓存中的元素
func (b *bigCache) delete(id int64) error {
    return b.users.Delete(cast.ToString(id))
}

// 读取缓存中的元素
func (b *bigCache) read(id int64) (cacheModel.User, error) {
    bs, err := b.users.Get(cast.ToString(id))
    if err != nil {
        if errors.Is(err, bigcache.ErrEntryNotFound) {
            return cacheModel.User{}, cacheModel.ErrUserNotInCache
        }

        return cacheModel.User{}, fmt.Errorf("get: %w", err)
    }

    var u cacheModel.User
    err = json.Unmarshal(bs, &u)
    if err != nil {
        return cacheModel.User{}, fmt.Errorf("unmarshal: %w", err)
    }
    return u, nil
}

func main() {
    b, err := newBigCache()
    if err != nil {
        fmt.Printf("new big cache err :%v", err)
    }
    u := cacheModel.User{
        Id:    1,
        Email: "1234@qq.com",
    }
    if err = b.update(u); err != nil {
        fmt.Printf("update big cache err :%v", err)
    }
    for {
        u, err := b.read(1)
        if err != nil {
            fmt.Printf("read big cache err :%v", err)
            return
        }
        fmt.Println(u)
        time.Sleep(1 * time.Second)
    }
}

结果：”LifeWindow: 5 * time.Second” 缓存数据为5秒钟，”CleanWindow: 2 * time.Second” 为每2秒检测一次过期元素并进行删除，上面运行结果每隔1秒获取一次元素并打印，则在第7秒时检测到元素已过期并进行删除，随后再次使用read()方法进行读取时，报错打印元素不存在。

read big cache err :the user is not in cache

我们使用JSON编码/解码元素，但也可以使用任何数据格式。例如，一种二进制格式Protobuf可以显著提高性能

  • 优点
    • 可用户生产环境
    • 丰富的缓存配置
    • 维护当中
    • 缓存不会触发GC，在大元素存储性能高
  • 缺点
    • 需要自己实现元素编解码。

3.总结

我们调研了Golang的不同内存缓存。记住没有最好的解决方案，需要根据实际应用场景来决定。 使用本文来比较解决方案，并决定哪一个适合您的项目需要
